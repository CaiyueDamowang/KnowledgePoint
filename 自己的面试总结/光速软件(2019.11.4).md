findHandler(method, url) {
    var obj = {
        "Get": "query",
        "Post": "add",
        "Delete": "delete",
        "Update": "update"
    }
    // 正则替换
    var reg = 'http://www.xxx.com/';
    var num = url.replace(function(reg) {
        return '';
    })
    
    // 替换头字母为大写   可用正则来做
    var len = num;
        newNum = '';
    for(var i = 0; i < len; i++) {
        if(i === 0) {
            newNum += i.toUpperCase();
        }else {
            newNum += i;
        }
    }
    
    return obj[method] + newNum;
}
